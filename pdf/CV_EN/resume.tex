\section{\texorpdfstring{\faFileText\ SELECTED PUBLICATIONS}{SELECTED PUBLICATIONS}}
\datedline{{\textbf{[AAAI'26]}} \textperiodcentered\ \textbf{FineXtrol: Controllable Motion Generation via Fine-Grained Text}}{}
\entrydetail{\textbf{\uline{Keming Shen}}, Bizhu Wu, Junliang Chen, Xiaoqin Wang, Linlin Shen \ [\href{https://arxiv.org/abs/2511.18927}{\faFilePdfO\ Paper Link}]}
\begin{itemize}
  \item Proposed FineXtrol, a controllable motion generation framework that replaces costly coordinate-sequence control with user-friendly fine-grained textual control signals.
  \item Designed a hierarchical contrastive learning module to improve the text encoder's discriminative ability for fine-grained motion descriptions, thereby enhancing generation quality.
  \item Experiments on the HumanML3D test set demonstrate strong performance in precise multi-body-part control and usability, indicating strong potential for downstream applications.
\end{itemize}

\datedline{\textbf{[CVPR'25]} \textperiodcentered\ MG-MotionLLM: A Unified Framework for Motion Comprehension and Generation across Multiple Granularities}{}
\entrydetail{Bizhu Wu, Jinheng Xie, \textbf{\uline{Keming Shen}}, Zhe Kong, Jianfeng Ren, Ruibin Bai, Rong Qu, Linlin Shen \ [\href{https://arxiv.org/abs/2504.02478}{\faFilePdfO\ Paper Link}]}
\begin{itemize}
  \item Proposed MG-MotionLLM, a unified framework that jointly models motion comprehension and generation tasks across multiple granularities.
  \item Developed a multi-granularity training scheme with auxiliary tasks to capture motion-related features at different granularity levels and strengthen comprehensive motion understanding.
  \item The model supports a range of new downstream tasks, including fine-grained motion editing, and enables cross-task generalization within a single unified model.
\end{itemize}

\datedline{\textbf{[NeurIPS'26 (Under Review)]} \textperiodcentered\ Ciallo: Coordinated and Controllable 3D Talking Avatar with Holistic \& Fine-Grained Face-Body Space}{}
\entrydetail{Yuhang Guo, \textbf{\uline{Keming Shen}}, Kaijun Deng, Siyang Song, Zehai Niu, Jinbao Wang, Linlin Shen \ [\href{https://openreview.net/attachment?id=AR3C3ya1i1&name=pdf}{\faFilePdfO\ Paper Link}]}
\begin{itemize}
  \item Proposed Ciallo, the first 3D talking avatar framework to unify gesture and facial expression generation at the representation level within a shared space.
  \item Designed a fine-grained body-part motion modeling strategy with spatiotemporally aware textual feature injection to enable coordinated and controllable motion generation for specific body parts.
  \item Built a scalable 3D talking avatar dataset from in-the-wild videos by combining a monocular tracker with VLM-based video understanding.
\end{itemize}
